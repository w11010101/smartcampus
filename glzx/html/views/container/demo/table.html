    el:"#tableVM",
    data:{
        tableHead:tableHead,
        arr:arr
    },
    computed:{

    },
    filters:{
        setSort:function(event){
            if(typeof event === "object"){
                return event.text;
            }else{
                return event;
            }
        }
    },
    methods:{
        sortFn:function(type,event,key){
            console.log(arguments);

            this.arr.sort(this.compareFn(type,'text',key));
        },
        // 比较方法
        compareFn:function(type,val,key){
            return function(a,b){
                var val1,val2;
                //判断当前是不是对象
                if(typeof a[key] === "string"){
                    val1 = a[key];
                    val2 = b[key];
                }else if(typeof a[key] === "object"){
                    val1 = a[key][val];
                    val2 = b[key][val];
                }
                // 统一比较的格式
                if(!isNaN(Number(val1)) && !isNaN(Number(val2))){
                    val1 = Number(val1);
                    val2 = Number(val2);
                }
                // 排序
                if(type === "ascend"){
                    // 升序
                    if(val1 < val2){
                        return -1;
                    }else if(val1 > val2){
                        return 1;
                    }else{
                        return 0;
                    }
                }else{
                    // 降序
                    if(val2 < val1){
                        return -1;
                    }else if(val2 > val1){
                        return 1;
                    }else{
                        return 0;
                    }
                }
            }
        }
    }
});
